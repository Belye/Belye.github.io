# 花式索引

print(arr[np.ix_([0,3,5],[0,1,3])]) # 索引0,3,5行的0,1,3列数据

print('----------------Tips-------------------------------')
print('通过花式索引获取的新的数组是一个副本,新数组修改不会改变原数组')
