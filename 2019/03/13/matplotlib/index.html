Matplotlib概述

Python最常用的绘图库,提供了一整套十分适合交互式绘图的命令API,比较方便的就可以将其嵌入到GUI应用程序中。






线图

figure使用

data = np.arange(-3,3,0.1)
x = np.sin(data)
y = np.cos(data)


# 绘图
plt.figure() #plt.figure(figsize = (8,6)) 这边是可以指定画布的大小的
plt.plot(data,x)
plt.show()

plt.figure() #plt.figure(figsize = (8,6)) 这边是可以指定画布的大小的
plt.plot(data,y)
plt.show()

颜色，标记，线型

x = [1,3,5]
y = [2,4,3]
plt.plot(x,y,color = 'r',linestyle = '--',marker = 'o',label = 'x-line')
plt.legend()
plt.show()

刻度，标题，标签，图例

x = [1,3,5]
y = [2,4,3]

# 创建图形
plt.figure()
# 画图
plt.plot(x,y,'r--',label = 'x-line')
#生成图例
plt.legend(loc = 'upper left')

#设置标题
plt.title('scores')

# 设置x，y标签
plt.xlabel('x axis')
plt.ylabel('y axis')

#设置刻度的范围

ax = plt.gca()

plt.show()

subplot子图

x1 = [1,3,5]
y1 = [2,4,5]

x2 = [1,4,5]
y2 = [3,2,5]

#创建图形
plt.figure()

plt.subplot(221)
plt.plot(x1,y1,'r--')

plt.subplot(222)
plt.plot(x2,y2,'b--')

plt.show()

子图面向对象的写法

x1 = [1,3,5]
y1 = [2,4,5]

x2 = [1,4,5]
y2 = [3,2,5]

#创建图形
fig = plt.figure()
ax1 = fig.add_subplot(221)
ax1.plot(np.random.randn(50).cumsum(),'r--')

ax2 = fig.add_subplot(222)
ax2.plot(np.random.randn(50).cumsum(),'b--')

ax3 = fig.add_subplot(223)
ax3.plot(np.random.randn(50).cumsum(),'y--')

ax4 = fig.add_subplot(224)
ax4.plot(np.random.randn(50).cumsum(),'g--')

plt.show()

subplot子图

# 生成一个figure以及子图列表
fig,ax = plt.subplots(2,2) # 2行2列 本质是一个列表

for i in range(2):
    for j in range(2):
        ax[i][j].plot(np.random.randn(50).cumsum(),'r--')

plt.savefig('子图')
plt.show()






柱状图

# 设置数据
height = np.array([100,200,300,400])

# 设置数据标量
left = np.arange(1,5,1) # 4 x

# 设置柱子宽度
n = 4
width = 0.8 / n

# 创建figure
plt.figure()

plt.title('City')

plt.xlabel('month')
plt.ylabel('money')

# 画图
plt.bar(left,height,width = width,color = 'lightskyblue',align = 'center',label = 'SH') 
plt.bar(left-width,height + 20,width = width,color = 'red',align = 'center',label = 'BJ') #第一个参数是设定偏移量
plt.bar(left+width,height + 10,width = width,color = 'y',align = 'center',label = 'SZ')
plt.legend()

plt.show()






直方图

mu, sigma = 100,15
x = np.random.randn(1000) * mu * sigma

#绘图
plt.figure()
plt.hist(x,100,alpha = 0.5,color = 'red') #第二个参数用来设定柱子个数 第三个参数用来设定颜色深浅

plt.grid(True) #加上网格

plt.show()






散点图

# 设置数据
x = np.random.randn(1000)
y = np.random.randn(1000)

# plt.scatter(x,y)
# plt.scatter(x,y,c = np.random.rand(1000))
# plt.scatter(x,y,c = np.random.rand(1000),s = np.random.rand(1000) * 50)
plt.scatter(x,y,c = np.random.rand(1000),s = np.random.rand(1000) * 50,alpha = 0.8)
plt.title('money')

plt.show()






饼图

# 设置数据
x = [15,25,30,30]

#设置颜色
color = ['r','yellow','green','blue']

#设置每一块离开中心的距离
explode = [0.2,0,0,0]

#每一块的文字说明
labels = ['yule','shenghuo','grilfriend','game']


plt.figure()




# 绘制饼图
plt.pie(x,explode = explode,labels = labels, colors = color,startangle = 90,shadow = True ,labeldistance = 1.2,autopct = '%.1f%%')

plt.title('money')
# 设置圆图
plt.axis('equal')
plt.show()
